select PERSONNELNUMBER workernumber,DIRADDRESSBOOK.NAME addrbook,  DIRPARTYTABLE.name from HcmWorker
join DIRADDRESSBOOKPARTYVIEW on
HcmWorker.PERSON = DIRADDRESSBOOKPARTYVIEW.PARTY
join DIRADDRESSBOOK on
DIRADDRESSBOOK.RECID = DIRADDRESSBOOKPARTYVIEW.ADDRESSBOOK
join DIRPARTYTABLE on
DIRPARTYTABLE.recid = HcmWorker.PERSON
